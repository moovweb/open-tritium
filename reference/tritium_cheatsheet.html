$(”/html/body”) {
	insert(”div”, id:”header”) {
		# Move the menu into my header
		move_here(”..//div[@id=’menu’]”) {
			remove(”./a[@class=’faqs’]”)
			$$(”li”) {
				log(index())
			}
		}
	}
}

# Replacing Text
$variable = “Example”

# Match/with are essentially if/then statements
match($variable) {
	with(/Sample/) {
		# Check the terminal to see your logs
		log(”Matched!”)
	}
	else() { log(”Failed.”) }
}
